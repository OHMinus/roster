28def capture_def(line: str, fh: type(open)) -> dict:
29 """ captures and parses a function definition
30
31 **LIMITATION:**
32 - no comment allowed within and at the end an function definition
33 - only pylint conform code
34 - sting context is not recognized (e.g. the character `->` or `,`, '=' or `:` may miss-interpreted)
35
36 **SUPPORTED FUNCTION STYLES (Examples):**
37 - `def func ( p1 : p1hint, pn : pnhint ) -> rhint :`
38 - `def func ( p1 : type(h1), pn : type(hn) ) -> type(hr) :`
39
40 **PROCEDURE STEPS:**
41 - (STEP-1) get definition: `def\s+(.*)\s*`
42 - (STEP-2) split func-with-params from return-hint: `(.*)->(.*)\\s*[:].*`
43 -> right part is the *return-hint*
44 - (STEP-3) split func-name and param-list: `(\\S+)\\s*[(]\\*(.*)\\s*[)]`
45 -> left part is the *function-name*
46 - (STEP-4) split param list: `split("\\s*[,]\\s*")`
47 - (STEP-5) split param from hint: `split("\\s*[:]\\s*")`
48
49 :param line: present line
50 :param fh: file-handle, points to next line
51 :return: dictionary containing information about the given subject or ( *None* = there is no 'def' keyword ) :
52 - 'indent' (str) = indentation (string of spaces) to the left of the function
53 - 'kind' (str) = kind of compound; always 'def'
54 - 'name' (str) = function name or ( *None* = there is no name )
55 - 'params' (list) = list of parameter names (in the right order); this include optional initializer-strings
56 - 'types' (dict) = types of parameters
57 """
58 res = None
59
60 # capture a function definition, even if it extends over several lines
61 rr = re.search(r'^\s*(def|class)', line)
62 if rr is not None:
63 while True:
64 rr = re.search(r'[:]\s*$', line)
65 if rr is not None:
66 break
67 line += fh.readline()
68
69 # parse function definition (keyword 'def')
70 # (STEP-1) get definition
71 # re.groups: 1:indent 2:kind 3:definition
72 rr = re.search(r'^(\s*)(def|class)\s+(.*)\s*$', line, flags=re.DOTALL | re.MULTILINE)
73 if rr is not None:
74 res = {'indent': rr.group(1), # 1: indention for function definition and doxygen text
75 'kind': rr.group(2), # 2: kind of compound definition (here always 'def')
76 'name': None, 'params': [], 'types': {'': ''}}
77 rest = rr.group(3) # 3: function definition
78
79 # (STEP-2a) split function-name from the rest
80 # re.groups: 1:func-name 2:rest
81 rr = re.match(r'^(\w+)\b\s*(.*)', rest, flags=re.DOTALL | re.MULTILINE)
82 if rr is not None:
83 res['name'] = rr.group(1) # 1: function name
84 rest = rr.group(2)
85
86 # (STEP-2b) split func-params from return-hint
87 # re.groups: 1:func-params 2:return-hint
88 rr = re.match(r'^([(]\s*.*\s*[)])\s*[-][>]\s*(.*)\s*[:].*', rest, flags=re.DOTALL | re.MULTILINE)
89 if rr is not None:
90 rest = rr.group(1) # 1: function parameters
91 res['types'][''] = rr.group(2) # 2: optional return type-hint
92
93 # (STEP-3) extract param-list
94 # re.groups: 1:params
95 rr = re.match(r'^[(]\s*(.*)\s*[)].*', rest, flags=re.DOTALL | re.MULTILINE)
96 if rr is not None:
97 rest = rr.group(1) # 2: parameter-list with optional type-hints (comma separated)
98 if rest is not None and rest is not '':
99 # (STEP-4) split param list
100 params = re.split(r'\s*[,]\s*', rest, flags=re.DOTALL | re.MULTILINE)
101
102 # (STEP-5) split param from hint
103 # .. NOTE: [''] is added to ensure two list elements if there is no type-hint
104 params2 = [] # [0]=name; [1]=type-hint; [2]=initializer
105 for pp in params:
106 pp2 = (re.split(r'\s*[=]\s*', pp) + [''])[:2] # separate initializer
107 pp2 = (re.split(r'\s*[:]\s*', pp2[0]) + [''])[:2] + [pp2[1]] # separate type-hint
108 params2.append(pp2)
109 # assign 'params' name with optional initializer; 'types' mapping name to type
110 res['params'] = []
111 for [a, b, c] in params2:
112 res['params'].append(a if c is '' else '{}={}'.format(a, c))
113 res['types'][a] = b
114 else:
115 pass # ignored: no parameters
116 else:
117 pass # (End Step-3) ignored: no parentheses
118 else:
119 pass # (End Step-2a) ignored: no function-name
120 else:
121 pass # (End Step-1) ignored: no compound definition
122 return res

125def capture_docstr(line: str, fh: type(open)) -> dict:
126 """captures and parses a doc-string
127
128 :param line: present line
129 :param fh: file-handle, points to next line
130 :return: dictionary containing information about the given subject or None:
131 - 'lines' = (list) docstring lines:
132 - 'indent' = (str) indentation within the docstring
133 - 'param' = (str) parameter name OR '' for return OR None if undefined
134 - 'text' = (str) remaining text
135 - 'types' = (dict) types captured from the doc-string
136 """
137 res = {'lines': [], 'types': {}}
138
139 # capture a doc-string, even if it extends over several lines
140 rr = re.search(r'^(\s*)[ru]*"""(.*)$', line)
141 if rr is not None:
142 line = rr.group(1) + rr.group(2) + '\n' # first line without leading triple-quotes
143 while True:
144 ll = {}
145 rr = re.search(r'^(.*)"""\s*$', line) # check for tailing triple-quotes
146 if rr is not None:
147 line = rr.group(1) + '\n' # (last) line without leading triple-quotes
148 rr2 = re.match(r'^(\s*)(.*)\n$', line) # separate indent from the rest
149 if rr2 is not None:
150 ll['indent'] = rr2.group(1)
151 ll['param'] = None
152 line = rr2.group(2)
153 rr2 = re.match(r'^[:](\w+[^:]*)[:]\s*(.*)', line) # check for ':param ...'/':return:'
154 if rr2 is not None:
155 g21 = rr2.group(1)
156 g22 = rr2.group(2)
157
158 rr3 = re.match(r'^[r]?type\s*(.*)\s*$', g21)
159 if rr3 is not None:
160 g31 = rr3.group(1)
161 res['types'][g31] = g22
162 line = None # suppress this line from output
163 else:
164 param = re.sub(r'param\s*', '', g21)
165 param = re.sub(r'return', '', param)
166 ll['param'] = param
167 line = g22
168 if line is not None:
169 ll['text'] = line
170 res['lines'].append(ll)
171 if rr is not None:
172 break
173 line = fh.readline()
174
175 return res

178def perform_fh(fh: type(open)) -> None:
179 """ perform filter on a given file handle
180
181 converts python code to a doxygen convenient format.
182
183 This contains the following work-around about the disadvantages at Doxygen v1.8.14:
184 - function definition may contain type hints;
185 - python docstrings are converted into regular doxygen format;
186 - type annotation is included into the doxygen format (from type hints XOR docstring types);
187
188 Output is written to standard output stream.
189 Error messages are written to standard error stream.
190
191 :param fh: input file handle
192 :return: None
193 """
194 sys.stdout = io.TextIOWrapper(sys.stdout.buffer,encoding='utf-8')
195 print('# generated by {} {}'.format(os.path.basename(__file__), __version__))
196 for line in fh:
197 by_pass = True
198
199 # capture function definition
200 fstr = capture_def(line, fh)
201 # do only if a function definition has found
202 if fstr is not None:
203 # ensure that function-name exists, otherwise put an error-message
204 if fstr['name'] is not None:
205 # only perform removing if there are type-hints
206 if any(list(fstr['types'].values())):
207 params = ', '.join(fstr['params'])
208 func_str = '{}{} {}({}): # type-hints removed'.format(fstr['indent'], fstr['kind'], fstr['name'],
209 params)
210 else:
211 func_str = re.sub(r'\n$', '', line) # no type-hints
212
213 line = fh.readline()
214
215 # capture doc-string
216 dstr = capture_docstr(line, fh)
217
218 # use the appropriate set-of-types
219 types = fstr['types']
220 if any(dstr['types'].values()) is True:
221 if any(fstr['types'].values()) is True:
222 print("{}: WARNING: type-hints are applied instead of docstring-types at '{} {}()'"
223 .format(fh.name, fstr['kind'], fstr['name']), file=sys.stderr)
224 else:
225 types = dstr['types']
226
227 # iterate over lines of docstring
228 for ii in range(len(dstr['lines'])):
229 comment = '#'
230 param = dstr['lines'][ii]['param']
231 text = dstr['lines'][ii]['text']
232 indent = dstr['lines'][ii]['indent']
233 if ii == 0:
234 comment = '##'
235 if param is None:
236 print('{}{} {}{}'.format(fstr['indent'], comment, indent, text))
237 else:
238 if param == '':
239 print(r'{}{} \return ({}) {}'.format(fstr['indent'], comment, types[param], text))
240 elif param in types:
241 print(r'{}{} \param {} ({}) {}'.format(fstr['indent'], comment, param, types[param], text))
242 else:
243 print("{}: WARNING: parameter '{}' unknown at '{} {}()' is not applied"
244 .format(fh.name, param, fstr['kind'], fstr['name']), file=sys.stderr)
245
246 print(func_str)
247 if len(dstr['lines']) != 0:
248 by_pass = False
249 else:
250 print("{}: ERROR: unable to capture '{}'".format(fh.name(), line.rstrip()), file=sys.stderr)
251 else:
252 pass # no function definition
253 if by_pass is True:
254 print(line, end='')

257def main(file: str) -> None:
258 """ filter main procedure
259
260 For further details cf. perform_fh().
261
262 :param file: input file name or '-' for input stream
263 :return:
264 """
265 if file == '-':
266 perform_fh(sys.stdin)
267 else:
268 try:
269 with open(file, 'r',encoding='utf-8') as fh:
270 perform_fh(fh)
271 except OSError as err:
272 print("ERROR: can't open '{}' for read: {} (errno={})".format(err.filename, err.strerror, err.errno),
273 file=sys.stderr)
274
275
276if __name__ == '__main__':
277 if len(sys.argv) >= (1+1):
278 main(sys.argv[1])
279 else:
280 pname = os.path.basename(__file__)
281 print('{} {}\n\nUSAGE: {} <name>.py'.format(pname, __version__, pname), file=sys.stderr)
